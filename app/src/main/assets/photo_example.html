JavaScript调用示例：


// 检查Android接口是否可用
if (window.AndroidPhoto) {
    // 打开相机
    AndroidPhoto.openCamera('camera_callback');

    // 打开相册
    AndroidPhoto.openGallery('gallery_callback');
} else {
    console.log('Android接口不可用，请在App内使用');
}

// 处理返回结果
window.handlePhotoResult = function(callbackId, imagePath) {
    console.log('回调ID:', callbackId, '图片路径:', imagePath);
    // 显示图片预览
    document.getElementById('imagePreview').src = imagePath;
    document.getElementById('imagePreview').style.display = 'block';
};

// 处理错误
window.handlePhotoError = function(callbackId, error) {
    console.error('错误:', error);
    alert('错误: ' + error);
};
